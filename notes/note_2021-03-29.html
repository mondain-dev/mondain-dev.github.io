\[ \DeclareMathOperator{\sech}{sech} \DeclareMathOperator{\sinh}{sinh} \DeclareMathOperator{\cosh}{cosh} \DeclareMathOperator{\diff}{d} \]

This integral is tweeted by N.N. Taleb (21 March, 2021) quoting 積分bot (21 March) \[ I = \int_0^1 \ln\ln \frac{1+\sqrt{1+x^2}}{x}\diff x \]

Let \(u\) denote the integrand, then \(x=\sech\exp u\) \[ \begin{align} I = \int_0^1 u \diff x & =\int_{-\infty}^\infty \frac{u e^u \exp(e^u) \left(\exp(2e^u)-1\right)}{(\exp(2e^u)+1)^2}\diff u \\ & = \int_{-\infty}^\infty \left.\frac{\partial}{\partial t}\right|_{t=1}\frac{-2\sinh(e^{u})}{\exp(2e^{tu})+1}\diff u \\ & = -2 \left.\frac{\partial}{\partial t}\right|_{t=1} \int_{-\infty}^\infty \frac{\sinh(e^{u})}{\exp(2e^{tu})+1}\diff u \\ & = -2 \left.\frac{\partial}{\partial t}\right|_{t=1} \int_{-\infty}^\infty \frac{1}{t} \frac{\sinh(e^{v/t})}{\exp(2e^{v})+1}\diff v \\ & = -2 \int_{-\infty}^\infty \left.\frac{\partial}{\partial t}\right|_{t=1} \frac{1}{t} \frac{\sinh(e^{v/t})}{\exp(2e^{v})+1}\diff v \\ & = 2 \int_{-\infty}^\infty \frac{\sinh(e^{v})}{\exp(2e^{v})+1}\diff v + 2 \int_{-\infty}^\infty \frac{v e^v\cosh(e^{v})}{\exp(2e^{v})+1}\diff v \\ & = 2 \int_{-\infty}^\infty \frac{\sinh(e^{v})}{\exp(2e^{v})+1}\diff v + \int_{-\infty}^\infty v e^v \exp(-e^{v}) \diff v. \end{align} \]

The first term \[ \begin{align} \int_{-\infty}^\infty \frac{\sinh(e^{v})}{\exp(2e^{v})+1}\diff v & = \int_{-\infty}^\infty \sum_{k=0}^{\infty}\frac{1}{(2k+1)!}\frac{e^{(2k+1)v}}{\exp(2e^{v})+1}\diff v \\ & = \sum_{k=0}^{\infty}\frac{1}{(2k+1)!}\int_{0}^\infty \frac{z^{2k}}{\exp(2z)+1} \diff z \\ & = \sum_{k=0}^{\infty}\frac{1}{(2k+1)!} 2^{-2k-1} \eta(2k+1)\Gamma(2k+1) \\ & = \sum_{k=0}^{\infty}\frac{1}{(2k+1)} 2^{-2k-1} \eta(2k+1) \\ & = \ln(2)/2 + \sum_{k=1}^{\infty}\frac{1}{(2k+1)} 2^{-2k-1} \eta(2k+1) \\ & = \ln(2)/2 + \sum_{k=1}^{\infty}\frac{1}{(2k+1)} 2^{-2k-1} \left(1-2^{-2k}\right)\zeta(2k+1) \\ & = \ln(2)/2 + \sum_{k=1}^{\infty}\frac{2^{-2k-1}}{(2k+1)} \zeta(2k+1) - \sum_{k=1}^{\infty}\frac{2^{-4k-1}}{(2k+1)}\zeta(2k+1) \\ & = \ln(2)/2 + \frac{1}{2}\sum_{k=2}^{\infty}\frac{\frac{1}{2}^{k} - \left(-\frac{1}{2}\right)^{k}}{k}\zeta(k) - \sum_{k=2}^{\infty}\frac{\frac{1}{4}^{k} - \left(-\frac{1}{4}\right)^{k}}{k}\zeta(k) \\ & = \ln(2)/2 + \frac{1}{2} \ln \Gamma\left(\frac{1}{2}\right) - \frac{\gamma}{4} - \frac{1}{2} \ln \Gamma\left(\frac{3}{2}\right) - \frac{\gamma}{4} - \ln \Gamma \left(\frac{3}{4}\right) + \frac{\gamma}{4} + \ln \Gamma\left(\frac{5}{4}\right) + \frac{\gamma}{4} \\ & = - \ln(2) + \ln \Gamma\left(\frac{1}{4}\right) - \ln \Gamma\left(\frac{3}{4}\right), \end{align} \]

and the second term

\[ \begin{align} \int_{-\infty}^\infty v e^v \exp(-e^{v}) \diff v = \int_{0}^\infty \ln z e^{-z} \diff z = -\gamma. \end{align} \]

Hence the integral \(I=-\gamma - 2\ln\frac{2\Gamma\left(\frac{3}{4}\right)}{\Gamma\left(\frac{1}{4}\right)}\).
